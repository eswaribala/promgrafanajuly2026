alertmanager:
  ## If false, alertmanager will not be installed
  ##
  enabled: true
  # Alertmanager HA replicas
  replicaCount: 3
  # Prefer placing replicas on different Kubernetes nodes.
  # Use "soft" when you have only one node.
  podAntiAffinity: soft
  podAntiAffinityTopologyKey: kubernetes.io/hostname

  # Keep two instances available during voluntary maintenance.
  podDisruptionBudget:
    minAvailable: 2

  config:
   global:
     resolve_timeout: 5m
   
   route:
      group_by: ['alertname']
      group_wait: 30s
      group_interval: 5m
      repeat_interval: 1h
      receiver: 'slack-notifications'   # default receiver
      routes:
        
        
        # Custom route for alerts from specific namespaces
        - match:
          severity: 'page'
          receiver: 'slack-notifications'
          group_wait: 1m
          group_interval: 10m
          repeat_interval: 1h
        
   receivers:
      
      - name: 'slack-notifications'
        slack_configs:
            - api_url: ''
              channel: ''
              send_resolved: true
              title: "{{ .CommonLabels.alertname }} - {{ .Status }}"
              text: |
               {{ range .Alerts }}
                 *Alert:* {{ .Annotations.summary }}
                 *Description:* {{ .Annotations.description }}
                 *Severity:* {{ .Labels.severity }}
                 *Source:* {{ .GeneratorURL }}
               {{ end }}  
      - name: 'pager-duty-notifications'
        pagerduty_configs:
        - service_key: ''  
   
   inhibit_rules:
      - source_matchers:
          - inhibit="true"
        target_matchers:
          - inhibited="true"
        equal: ["instance", "job"]   
      
   
  persistence:
    size: 2Gi

  podSecurityContext:
    runAsUser: 65534
    runAsNonRoot: true
    runAsGroup: 65534
    fsGroup: 65534
